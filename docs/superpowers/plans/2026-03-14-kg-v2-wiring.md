# KG v2 Wiring Implementation Plan

> **For agentic workers:** REQUIRED: Use superpowers:subagent-driven-development (if subagents available) or superpowers:executing-plans to implement this plan. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Wire the KG v2 pipeline end-to-end: extract-graphiti endpoint, ChatServiceV2, saga profiles CRUD API, and Leiden community clustering.

**Architecture:** Four independent features wired into the existing FastAPI + arq infrastructure. The extract-graphiti endpoint enqueues the existing `process_book_graphiti` worker task. ChatServiceV2 wraps the existing `build_chat_v2_graph()`. Saga profiles are stored in Redis as JSON. Leiden clustering runs as a post-processing step in the worker.

**Tech Stack:** FastAPI, arq, Redis, Neo4j GDS (Leiden), LangGraph, Gemini 2.5 Flash.

**Spec:** `docs/superpowers/specs/2026-03-14-kg-v2-wiring-design.md`

```python
# backend/app/api/routes/saga_profiles.py
"""Saga profile CRUD API routes."""
from __future__ import annotations

from fastapi import APIRouter, Depends, Request
from fastapi.responses import JSONResponse

from app.api.auth import require_auth
from app.core.logging import get_logger
from app.schemas.saga_profile import (
    SagaProfileListItem,
    SagaProfileListResponse,
    SagaProfileResponse,
)
from app.services.saga_profile.models import SagaProfile

logger = get_logger(__name__)
router = APIRouter(prefix="/saga-profiles", tags=["saga-profiles"])

REDIS_PREFIX = "saga_profile:"


async def _get_redis(request: Request):
    return request.app.state.redis


@router.get("", response_model=SagaProfileListResponse)
async def list_profiles(redis=Depends(_get_redis)) -> SagaProfileListResponse:
    """List all saga profiles."""
    keys = await redis.keys(f"{REDIS_PREFIX}*")
    items = []
    for key in keys:
        raw = await redis.get(key)
        if raw:
            profile = SagaProfile.model_validate_json(raw)
            items.append(SagaProfileListItem(
                saga_id=profile.saga_id,
                saga_name=profile.saga_name,
                version=profile.version,
                entity_types_count=len(profile.entity_types),
                patterns_count=len(profile.text_patterns),
                estimated_complexity=profile.estimated_complexity,
            ))
    return SagaProfileListResponse(profiles=items, total=len(items))


@router.get("/{saga_id}", response_model=SagaProfileResponse)
async def get_profile(saga_id: str, redis=Depends(_get_redis)) -> SagaProfileResponse:
    """Get a saga profile by ID."""
    raw = await redis.get(f"{REDIS_PREFIX}{saga_id}")
    if raw is None:
        return JSONResponse(status_code=404, content={"detail": "Saga profile not found"})
    profile = SagaProfile.model_validate_json(raw)
    return SagaProfileResponse(profile=profile)


@router.put("/{saga_id}", response_model=SagaProfileResponse)
async def put_profile(
    saga_id: str,
    body: SagaProfile,
    redis=Depends(_get_redis),
) -> SagaProfileResponse:
    """Create or update a saga profile."""
    await redis.set(f"{REDIS_PREFIX}{saga_id}", body.model_dump_json())
    logger.info("saga_profile_saved", saga_id=saga_id)
    return SagaProfileResponse(profile=body)


@router.delete("/{saga_id}")
async def delete_profile(saga_id: str, redis=Depends(_get_redis)):
    """Delete a saga profile."""
    deleted = await redis.delete(f"{REDIS_PREFIX}{saga_id}")
    if not deleted:
        return JSONResponse(status_code=404, content={"detail": "Saga profile not found"})
    logger.info("saga_profile_deleted", saga_id=saga_id)
    return {"deleted": True, "saga_id": saga_id}
```

```python
from app.schemas.saga_profile import ExtractGraphitiRequest

@router.post(
    "/{book_id}/extract-graphiti",
    dependencies=[Depends(require_auth)],
    status_code=202,
)
async def extract_graphiti(
    book_id: str,
    body: ExtractGraphitiRequest,
    http_request: Request,
    arq_pool=Depends(get_arq_pool),
):
    """Trigger Graphiti KG v2 extraction for a book.

    Discovery Mode (first book in saga): extracts with universal types,
    then auto-induces a SagaProfile.

    Guided Mode (subsequent books): uses existing SagaProfile from Redis.
    """
    redis = http_request.app.state.redis
    existing_profile = await redis.get(f"saga_profile:{body.saga_id}")
    mode = "guided" if existing_profile else "discovery"

    job = await arq_pool.enqueue_job(
        "process_book_graphiti",
        book_id,
        body.saga_id,
        body.saga_name,
        body.book_num,
        existing_profile,  # None for discovery, JSON string for guided
        _queue_name="worldrag:arq",
        _job_id=f"graphiti:{book_id}",
    )
    logger.info(
        "extract_graphiti_enqueued",
        book_id=book_id,
        saga_id=body.saga_id,
        mode=mode,
    )
    return {"job_id": job.job_id, "mode": mode, "book_id": book_id}
```

```python
# backend/app/services/chat_service_v2.py
"""Chat service v2 — Graphiti-based retrieval."""
from __future__ import annotations

import json
from typing import TYPE_CHECKING, Any

from langchain_core.messages import AIMessageChunk, HumanMessage

from app.agents.chat_v2.graph import build_chat_v2_graph
from app.core.logging import get_logger
from app.schemas.chat import ChatResponse, Citation, RelatedEntity, SourceChunk

if TYPE_CHECKING:
    from collections.abc import AsyncGenerator

    from neo4j import AsyncDriver

    from app.core.graphiti_client import GraphitiClient

logger = get_logger(__name__)


class ChatServiceV2:
    """Chat service using Graphiti retrieval + chat_v2 LangGraph pipeline."""

    def __init__(
        self,
        graphiti: GraphitiClient,
        neo4j_driver: AsyncDriver,
        checkpointer: Any = None,
    ) -> None:
        builder = build_chat_v2_graph(graphiti=graphiti, neo4j_driver=neo4j_driver)
        self._graph = builder.compile(checkpointer=checkpointer)

    async def query(
        self,
        query: str,
        book_id: str,
        saga_id: str,
        *,
        max_chapter: int | None = None,
        thread_id: str | None = None,
        **kwargs: Any,
    ) -> ChatResponse:
        """Run the v2 chat pipeline."""
        state_input: dict[str, Any] = {
            "messages": [HumanMessage(content=query)],
            "query": query,
            "original_query": query,
            "book_id": book_id,
            "saga_id": saga_id,
            "max_chapter": max_chapter,
            "retries": 0,
        }
        config: dict[str, Any] = {}
        if thread_id:
            config["configurable"] = {"thread_id": thread_id}

        result = await self._graph.ainvoke(state_input, config=config)

        answer = result.get("generation", "I wasn't able to generate an answer.")
        context = result.get("retrieved_context", [])
        gen_output = result.get("generation_output", {})

        logger.info(
            "chat_v2_query_completed",
            book_id=book_id,
            saga_id=saga_id,
            context_items=len(context),
            faithfulness=result.get("faithfulness_score"),
        )

        return ChatResponse(
            answer=answer,
            sources=[],
            related_entities=[],
            chunks_retrieved=len(context),
            chunks_after_rerank=len(context),
            thread_id=thread_id,
            citations=[],
            confidence=result.get("faithfulness_score", 0.0),
            entities_mentioned=[],
        )

    async def query_stream(
        self,
        query: str,
        book_id: str,
        saga_id: str,
        *,
        max_chapter: int | None = None,
        thread_id: str | None = None,
        **kwargs: Any,
    ) -> AsyncGenerator[dict[str, str], None]:
        """Stream the v2 pipeline as SSE events."""
        state_input: dict[str, Any] = {
            "messages": [HumanMessage(content=query)],
            "query": query,
            "original_query": query,
            "book_id": book_id,
            "saga_id": saga_id,
            "max_chapter": max_chapter,
            "retries": 0,
        }
        config: dict[str, Any] = {}
        if thread_id:
            config["configurable"] = {"thread_id": thread_id}

        try:
            async for stream_type, event_data in self._graph.astream(
                state_input, config=config, stream_mode=["messages", "custom"]
            ):
                if stream_type == "custom":
                    yield {
                        "event": "step",
                        "data": json.dumps(event_data if isinstance(event_data, dict) else {}),
                    }
                elif stream_type == "messages":
                    chunk_msg, _ = event_data
                    if isinstance(chunk_msg, AIMessageChunk) and chunk_msg.content:
                        content = chunk_msg.content
                        token = content if isinstance(content, str) else str(content)
                        yield {"event": "token", "data": json.dumps({"token": token})}
        except Exception as exc:
            logger.exception("chat_v2_stream_error")
            yield {"event": "error", "data": json.dumps({"message": f"Stream failed: {type(exc).__name__}"})}
            return

        yield {"event": "done", "data": "{}"}
```

```python
from __future__ import annotations

    from neo4j import AsyncDriver

logger = get_logger(__name__)


async def _summarize_community(names: list[str], summaries: list[str]) -> str:
    """Generate a one-paragraph summary for a community of entities."""
    from app.config import settings
    from app.llm.providers import get_llm

    entities_text = "\n".join(
        f"- {name}: {summary}" for name, summary in zip(names, summaries) if summary
    )
    prompt = (
        f"Summarize this group of entities from a fiction novel in 2-3 sentences. "
        f"Focus on what connects them.\n\n{entities_text}"
    )
    try:
        llm = get_llm(settings.llm_generation)
        response = await llm.ainvoke(prompt)
        return response.content if hasattr(response, "content") else str(response)
    except Exception:
        logger.warning("community_summary_llm_failed")
        return f"Community of {len(names)} entities: {', '.join(names[:5])}"


async def run_community_clustering(
    driver: AsyncDriver,
    saga_id: str,
) -> dict[str, Any]:
    """Run Leiden clustering on Graphiti entities and generate community summaries.

    1. Project Entity nodes + RELATES_TO edges via GDS
    2. Run Leiden algorithm
    3. Fetch communities
    4. Generate LLM summaries
    5. Store as :Community nodes
    """
    try:
        async with driver.session() as session:
            # Step 1: Project graph
            await session.run(
                """
                CALL gds.graph.project('saga-community', {
                    Entity: { properties: ['group_id'] }
                }, {
                    RELATES_TO: { orientation: 'UNDIRECTED' }
                })
                """
            )

            # Step 2: Run Leiden
            await session.run(
                """
                CALL gds.leiden.write('saga-community', {
                    writeProperty: 'community_id',
                    includeIntermediateCommunities: false
                })
                """
            )

            # Step 3: Fetch communities
            result = await session.run(
                """
                MATCH (n:Entity {group_id: $saga_id})
                WHERE n.community_id IS NOT NULL
                WITH n.community_id AS cid, collect(n.name) AS names, collect(n.summary) AS summaries
                WHERE size(names) >= 2
                RETURN cid AS community_id, names, summaries
                ORDER BY size(names) DESC
                """,
                saga_id=saga_id,
            )
            communities = await result.data()

            # Step 4: Generate summaries and store
            for community in communities:
                summary = await _summarize_community(
                    community["names"], community["summaries"]
                )
                await session.run(
                    """
                    MERGE (c:Community {community_id: $cid, saga_id: $saga_id})
                    SET c.summary = $summary,
                        c.member_count = $member_count,
                        c.members = $members
                    """,
                    cid=community["community_id"],
                    saga_id=saga_id,
                    summary=summary,
                    member_count=len(community["names"]),
                    members=community["names"][:20],
                )

            # Cleanup projection
            await session.run("CALL gds.graph.drop('saga-community', false)")

            logger.info(
                "community_clustering_completed",
                saga_id=saga_id,
                communities=len(communities),
            )
            return {"communities_found": len(communities), "saga_id": saga_id}

    except Exception as e:
        logger.warning("community_clustering_failed", error=type(e).__name__, saga_id=saga_id)
        # Try cleanup
        try:
            async with driver.session() as session:
                await session.run("CALL gds.graph.drop('saga-community', false)")
        except Exception:
            pass
        return {"communities_found": 0, "saga_id": saga_id, "error": str(e)}
```
